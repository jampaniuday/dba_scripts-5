create or replace procedure sp_auto_off is
cursor c1 is
select 'ALTER DATABASE DATAFILE '''||FILE_NAME||''' AUTOEXTEND OFF' TEXTO 
from dba_data_files where autoextensible = 'YES';
begin
	for tmp in c1 loop
		EXECUTE IMMEDIATE TMP.TEXTO;	
	end loop;
end;
/
