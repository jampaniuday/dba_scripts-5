select a.SEC_ACTUACION "ACTUACION", trunc(rtrim(a.NUM_TELEFONO)) "TELEFONO", b.DES_SBACTUACION "Actuacion Realizada", substr(c.DES_ESTAREC,0,9) "Estado Actuacion", a.FEC_ACTUACION "FECHA", trunc(a.IMP_COSTE/100) "Coste", trunc(a.IMP_SALDO/100) "Saldo",  a.COD_USUARIO "USUARIO" from ppga_actabopre a, ppga_literalsbactuacion b, PPGA_ESTAREC c where a.num_telefono = '$1' and a.COD_ACTUACIO = b.COD_ACTUACIO and a.COD_SBACTUACION = b.COD_SBACTUACION and b.COD_IDIOMA = '10' and a.cod_estarec = c.cod_estarec order by 1;
